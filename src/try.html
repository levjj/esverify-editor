function max(a, b) {
  requires(typeof(a) === 'number');
  requires(typeof(b) === 'number');
  ensures(max(a, b) >= a);
  ensures(max(a, b) >= b);

  if (a >= b) {
    return a;
  } else {
    return a;
  }
}
